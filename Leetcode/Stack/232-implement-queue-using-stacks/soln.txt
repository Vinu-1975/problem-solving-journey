class MyQueue:

    def __init__(self):
        self.stack1 = []
        self.stack2 = []
        

    def push(self, x: int) -> None:
        self.stack1.append(x)

    # for every 1st pop, its O(n), remaining time its O(1), similar in the case of peek()
    def pop(self) -> int:
        if self.stack2:
           return self.stack2.pop()
        else:
            while self.stack1:
                self.stack2.append(self.stack1.pop())
            return self.stack2.pop()

    def peek(self) -> int:
        if self.stack2:
            return self.stack2[-1]
        else:
            while self.stack1:
                self.stack2.append(self.stack1.pop())
            return self.stack2[-1]
        

    def empty(self) -> bool:
        return len(self.stack1) == 0 and len(self.stack2) == 0
