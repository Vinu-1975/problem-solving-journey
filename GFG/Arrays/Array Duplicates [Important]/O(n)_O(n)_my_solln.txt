
from typing import List


class Solution:
    def duplicates(self, n : int, arr : List[int]) -> List[int]:
        # code here
        freq = {}
        res = []
        for num in arr:
            freq[num] = freq.get(num,0) + 1
            if freq[num] == 2:
                res.append(num)
        
        return res if res else [-1]
