def findDuplicates(arr, n):
    # Iterate through each element in the array
    for i in range(n):
        # Use element value to calculate the index
        index = arr[i] % n
        # Increment the value at the calculated index by n
        arr[index] += n
    
    # Prepare a list to store duplicates
    duplicates = []
    
    # Check which elements are duplicates
    for i in range(n):
        # If the value at index i is >= 2*n, it means the index i was visited more than once
        if arr[i] // n > 1:
            duplicates.append(i)
    
    # If no duplicates are found, return [-1]
    if not duplicates:
        return [-1]
    
    return sorted(duplicates)
